// Add a review
        public void AddReview(Review review) { 
        
            Context.Reviews.Add(review);
            Context.SaveChanges();
        
        }

        // Select * reviews
        public List<Review> GetReviews() {
            return Context.Reviews.AsNoTracking().ToList();
        }

        // Update a review

        public void UpdateReview(int index)
        {
            var reviewToUpdate = Context.Reviews.FirstOrDefault(m => m.ReviewID == index);
            if (reviewToUpdate != null)
            {

                Context.Reviews.Update(reviewToUpdate);
                Context.SaveChanges();
            }
        }

        // Delete a review

        public void DeleteReview(int index)
        {
            var reviewToDelete = Context.Reviews.FirstOrDefault(m => m.ReviewId == index);
            if (reviewToDelete != null) 
            {
                Context.Reviews.Remove(reviewToDelete);
                Context.SaveChanges();
            } 
        
        }
